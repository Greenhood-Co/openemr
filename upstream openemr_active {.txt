upstream openemr_active {
    server 127.0.0.1:18081;
}

# HTTP → HTTPS (Certbot HTTP-01 needs /.well-known if you use webroot renewals)
server {
    listen 80; 
    server_name openemr.greenhood.com.ng;

    location ^~ /.well-known/acme-challenge/ {
        root /var/www/html;
    }

    location / {
        return 301 https://$host$request_uri;
    }
}

server {
    listen 443 ssl http2; 
    server_name openemr.greenhood.com.ng;

    ssl_certificate /etc/letsencrypt/live/openemr.greenhood.com.ng/fullchain.pem;
    ssl_certificate_key /etc/letsencrypt/live/openemr.greenhood.com.ng/privkey.pem;
    include /etc/letsencrypt/options-ssl-nginx.conf;
    ssl_dhparam /etc/letsencrypt/ssl-dhparams.pem;

    client_max_body_size 64M;

    location / {
        proxy_pass http://openemr_active;
        proxy_http_version 1.1;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
        proxy_set_header X-Forwarded-Host $host;
        proxy_set_header Connection "";
        proxy_buffering off;
        proxy_redirect off;
        proxy_read_timeout 300;
    }
}
